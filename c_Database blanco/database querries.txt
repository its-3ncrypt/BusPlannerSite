create table users (
    ID_User int PRIMARY KEY AUTO_INCREMENT,
    Username varchar(100),
    Password varchar(100),
    Voornaam varchar(100),
    Naam varchar(100),
    Email varchar(100),
    Bevestigd boolean,
    Rol varchar(100),
    ID_Afdeling int);

create table Afdelingen (
    ID_Afdeling int PRIMARY KEY AUTO_INCREMENT,
    Naam varchar(100));

create table bussen (
    ID_Bus int PRIMARY KEY AUTO_INCREMENT,
    Nummerplaat varchar(100),
    status boolean);

create table reservaties (
    ID_Reservaties int PRIMARY KEY AUTO_INCREMENT,
    Startdatum date,
    Stopdatum date,
    Startuur time,
    Stopuur time,
    ID_User int,
    ID_Bus int,
    Controle boolean,
    Geweigerd boolean);

create table Berichten (
    ID_Bericht int PRIMARY KEY AUTO_INCREMENT,
    ID_User int,
    bericht varchar(1000),
    Datum date);
